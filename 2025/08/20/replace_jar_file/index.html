Python Script to Replace Files in JAR

Created2025-08-20|Updated2025-08-23
|Post Views:

This Python script helps developers efficiently replace a specific file inside a JAR and automatically update its modification timestamp. It is useful for automated deployments, patch updates, or modifying class files without rebuilding the entire JAR.

Core Function: replace_file_in_jar

  • File Replacement: Replace a target file inside the JAR with a new file.
  • Timestamp Update: Updates the internal modification time of the replaced file.
  • Backup Option: Creates a timestamped backup of the original JAR before replacement.
  • Detailed Logs: Prints English logs showing progress and operations.

Python Implementation

import zipfile, os, datetime, shutil

def replace_file_in_jar(jar_path, new_file_path, target_full_path, output_jar=None, backup=True):
    if not output_jar:
        output_jar = jar_path + ".modified.jar"

    now = datetime.datetime.now().timetuple()[:6]

    print(f"[INFO] Processing JAR: {jar_path}")
    with zipfile.ZipFile(jar_path, 'r') as jar_in, zipfile.ZipFile(output_jar, 'w', zipfile.ZIP_DEFLATED) as jar_out:
        replaced = False
        for item in jar_in.infolist():
            if item.filename == target_full_path:
                print(f"[REPLACE] {item.filename}")
                info = zipfile.ZipInfo(item.filename, date_time=now)
                with open(new_file_path, 'rb') as f:
                    jar_out.writestr(info, f.read())
                replaced = True
            else:
                jar_out.writestr(item, jar_in.read(item.filename))
        if not replaced:
            raise FileNotFoundError(f"{target_full_path} not found in JAR")

    if backup:
        timestamp = datetime.datetime.now().strftime("%Y%m%d%H%M%S%f")[:-3]
        dir_name, base_name = os.path.split(jar_path)
        name, ext = os.path.splitext(base_name)
        backup_path = os.path.join(dir_name, f"{name}{timestamp}{ext}")
        shutil.copy2(jar_path, backup_path)
        print(f"[BACKUP] Original JAR backed up at: {backup_path}")

    os.replace(output_jar, jar_path)
    print(f"[SUCCESS] Replaced {target_full_path} and updated timestamp")

# Example usage
if __name__ == "__main__":
    replace_file_in_jar(
        jar_path="utils.jar",
        new_file_path="ZooApplication.class",
        target_full_path="BOOT-INF/classes/com/azusr/zoo/ZooApplication.class"
    )
